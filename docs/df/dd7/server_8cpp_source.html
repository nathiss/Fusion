121  const auto make_game_full = [](std::size_t id){
122  return JSON({
123  {"id", {id}},
124  {"result", {"full"}},
125  }, false, JSON::value_t::object);
126  };
127 
128  if (request["type"] == "join") {
129  std::string game_name = request["game"];
130  games_mtx_.lock();
131  auto it = games_.find(game_name);
132  if (it == games_.end()) {
133  it = games_.emplace(game_name, std::make_shared<Game>(game_name)).first;
134  }
135  games_mtx_.unlock();
136  auto join_result = it->second->Join(src);
137  if (!join_result) { // The game is full.
138  return std::make_pair(false, make_game_full(request["id"]));
139  }
140  // If we're here it means the join was successful.
141  src->delegate_ = std::get<0>(join_result.value());
142 
143  {
144  std::lock_guard l{unidentified_sessions_mtx_};
145  unidentified_sessions_.erase(src);
146  }
147  {
148  std::lock_guard l{sessions_correlation_mtx_};
149  sessions_correlation_[src] = std::make_optional<std::string>(std::move(game_name));
150  }
151 
152  auto response = [&request, &join_result] {
153  return JSON({
154  {"id", {request["id"]}},
155  {"result", {"joined"}},
156  {"my_id", {std::get<2>(join_result.value())}},
157  {"players", {std::get<1>(join_result.value())["players"]}},
158  {"rays", {std::get<1>(join_result.value())["rays"]}},
