72 void WebSocketSession::Close() noexcept {
73  if (!in_closing_procedure_) {
74  std::lock_guard l{outgoing_queue_mtx_};
75  in_closing_procedure_ = true;
76  if (outgoing_queue_.size() > 1) {
77  // This means there are queued additional packages. We return and allow
78  // HandleWrite to call this method after the writing has been completed.
79  outgoing_queue_.erase(outgoing_queue_.begin() + 1, outgoing_queue_.end());
80  return;
81  }
82  if (outgoing_queue_.size() == 1) {
83  // This means there are no additional packages. We just returnand allow
84  // HandleWrite to call this method after the writing has been completed.
85  return;
86  }
87  }
88 
89  // If we're here it means either there are no packages queued or we are
90  // already in closing procedure.
91 
92  boost::system::error_code ec;
93  websocket_.close(boost::beast::websocket::close_code::none, ec);
94  // Now callers should not performs writing to the WebSocket and should
95  // perform reading as long as a read returns boost::beast::websocket::closed
96  // error.
97 
98  if (ec) {
99  logger_->warn("An error occured during closing a websocket. [Boost: {}]",
100  ec.message());
101  // TODO: Find out if the comment below applies to the WebSocket connections.
102  // We do nothing, because
103  // https://www.boost.org/doc/libs/1_67_0/doc/html/boost_asio/reference/basic_stream_socket/close/overload1.html
104  // [Set on failure. Note that, even if the function indicates an error, the underlying descriptor is closed.]
105  }
106 }

108 void WebSocketSession::Close(Package package) noexcept {
109  // We lock the mutex first, to ensure that no additional package will be
110  // queued, after the closing procedure has started.
111  std::lock_guard l{outgoing_queue_mtx_};
112  in_closing_procedure_ = true;
113 
114  if (outgoing_queue_.size() > 1) {
115  // This means we're already writing and other packages are waiting.
116  // We remove all additional packages and queue the closing package.
117  outgoing_queue_.erase(outgoing_queue_.begin() + 1, outgoing_queue_.end());
118  outgoing_queue_.push_back(package);
119  return;
120  }
121  if (outgoing_queue_.size() == 1) {
122  // This means we're already writing and no packages are waiting.
123  // We queue the closing package and return. HandleWrite method will perform
124  // sending and closing.
125  outgoing_queue_.push_back(package);
126  return;
127  }
128 
129  // If we're here it means no writing is performed right now.
130  boost::system::error_code ec;
131  websocket_.write(boost::asio::buffer(*package), ec);
132  if (ec) {
133  logger_->warn("An error occured during sync writing the closing message. [Boost:{}]",
134  ec.message());
135  }
136  Close();
137 }

244  logger_->debug("Written {} bytes to {}.", bytes_transmitted,
