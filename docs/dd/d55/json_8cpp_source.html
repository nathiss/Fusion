107  if (!parsed) {

111  auto json = std::move(parsed.value());
112 
113  if (!(json.contains("type") &&
114  json["type"].type() == decltype(json)::value_t::string)) {
115  return std::make_pair(false, MakeTypeNotFound());
116  }
117 
118  if (json["type"] == "join") {

120  json.contains("nick") &&
121  json.contains("game") &&
122  json.size() == 3 + 1 &&
