88 void Listener::HandleAccept(const boost::system::error_code& ec) noexcept {
89  if (ec == boost::asio::error::already_open) {
90  // This means the socket used to handle new connections in already open.
91  // We sacrifice one connection to make the acceptor work properly.
92  logger_->warn("The socket used to accept new connections is already open.");
93  socket_.close();
94  }
95 
96  if (ec == boost::asio::error::no_recovery) {
97  logger_->error("A non-recoverable error occurred during handling a new connection. [Boost:{}]",
98  ec.message());
99  return;
100  }
101  if (ec) {
102  logger_->error("An error occurred during handling a new connection. [Boost:{}]",
103  ec.message());
104  // TODO(nathiss): find out if any other error can cause the listener to stop working.
105  } else {

107  number_of_connections_++;

148  return false;
