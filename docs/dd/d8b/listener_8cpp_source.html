64  socket_.close();

66 
67  if (ec == boost::asio::error::no_recovery) {

93 
94  acceptor_.open(endpoint_.protocol(), ec);
95  if (ec) {
96  logger_->error("Open: {}", ec.message());
97  return;
98  }
99 
100  acceptor_.set_option(boost::asio::socket_base::reuse_address(true), ec);
101  if (ec) {

103  return;
104  }
105 
106  acceptor_.bind(endpoint_, ec);
107  if (ec == boost::asio::error::access_denied) {
108  // This means we don't have permision to bind acceptor to the this endpoint.
109  logger_->critical("Cannot bind acceptor to {} (permition denied).",
110  endpoint_);
111  return;
112  }
113  if (ec == boost::asio::error::address_in_use) {
114  // This means the endpoint is already is use.
115  logger_->critical("Cannot bind acceptor to {} (address in use).",
116  endpoint_);
117  return;
118  }
119  if (ec) {
120  logger_->error("Bind: {}", ec.message());
121  return;
122  }

125  if (ec) {

129

132 }
